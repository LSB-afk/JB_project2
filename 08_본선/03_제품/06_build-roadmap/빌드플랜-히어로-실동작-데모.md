**Goal:** 정적 MVP(`02_제품/app/`)를 그대로 두되, **전세(jeonse) 히어로 도메인 1개**만 실 공개 API(국토부 실거래가) 라이브 + 로컬모델 실추론으로 "진짜 도는" 데모로 승격. 나머지 도메인·감사체인은 현행 유지(감사체인은 이미 실제 해시체인·영속).

**Tech Stack:** Vanilla JS(ES2020, no modules), fetch/async, Node http(프록시), Ollama(EXAONE/Qwen), Playwright(E2E 검증), `verify_static.py`(문자열 계약).

```javascript
import http from "node:http";
const KEY = process.env.DATA_GO_KR_KEY;
const UP = "https://apis.data.go.kr/1613000/RTMSDataSvcAptRent/getRTMSDataSvcAptRent";
http.createServer(async (req, res) => {
  res.setHeader("Access-Control-Allow-Origin", "*");           // 데모 로컬 전용
  const u = new URL(req.url, "http://x");
  if (u.pathname !== "/jeonse") { res.writeHead(404).end(); return; }
  if (!KEY) { res.writeHead(503).end(JSON.stringify({ error: "no key" })); return; }
  try {
    const q = new URLSearchParams({ serviceKey: KEY, LAWD_CD: u.searchParams.get("lawd"), DEAL_YMD: u.searchParams.get("ym"), numOfRows: "50" });
    const r = await fetch(`${UP}?${q}`);
    const xml = await r.text();
    const deposits = [...xml.matchAll(/<deposit>\s*([\d,]+)\s*<\/deposit>/g)].map(m => Number(m[1].replace(/,/g, "")) * 10000);
    const estimatedPrice = deposits.length ? Math.round(deposits.reduce((a, b) => a + b, 0) / deposits.length) : 0;
    res.writeHead(200, { "content-type": "application/json" });
    res.end(JSON.stringify({ estimatedPrice, samples: deposits.length, source: "국토부 전월세 실거래가" }));
  } catch (e) { res.writeHead(502).end(JSON.stringify({ error: String(e) })); }
}).listen(8020, () => console.log("[proxy] :8020"));
```
